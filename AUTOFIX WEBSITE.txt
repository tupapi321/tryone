    <script>
        // JavaScript for responsive navigation toggle
        const navToggle = document.getElementById('nav-toggle');
        const navContent = document.getElementById('nav-content');

        navToggle.addEventListener('click', () => {
            navContent.classList.toggle('hidden');
        });

        // Close nav on link click for small screens
        const navLinks = navContent.querySelectorAll('a');
        navLinks.forEach(link => {
            link.addEventListener('click', () => {
                if (!navContent.classList.contains('hidden') && window.innerWidth < 1024) {
                    navContent.classList.add('hidden');
                }
            });
        });

        // AI Diagnosis Modal Logic
        const openAiDiagnosisModalBtn = document.getElementById('open-ai-diagnosis-modal');
        const aiDiagnosisModal = document.getElementById('ai-diagnosis-modal');
        const closeAiDiagnosisModalBtn = document.getElementById('close-ai-diagnosis-modal');
        const carIssueDescriptionInput = document.getElementById('car-issue-description');
        const getDiagnosisButton = document.getElementById('get-diagnosis-button');
        const diagnosisOutput = document.getElementById('diagnosis-output');
        const loadingSpinner = document.getElementById('loading-spinner');

        openAiDiagnosisModalBtn.addEventListener('click', () => {
            aiDiagnosisModal.classList.remove('hidden');
            diagnosisOutput.innerHTML = ''; // Clear previous diagnosis
            carIssueDescriptionInput.value = ''; // Clear input
        });

        closeAiDiagnosisModalBtn.addEventListener('click', () => {
            aiDiagnosisModal.classList.add('hidden');
        });

        // Close modal if clicking outside
        aiDiagnosisModal.addEventListener('click', (event) => {
            if (event.target === aiDiagnosisModal) {
                aiDiagnosisModal.classList.add('hidden');
            }
        });

        getDiagnosisButton.addEventListener('click', async () => {
            const prompt = carIssueDescriptionInput.value.trim();
            if (!prompt) {
                diagnosisOutput.innerHTML = '<p class="text-red-500">Please describe your car issue.</p>';
                return;
            }

            loadingSpinner.classList.remove('hidden');
            diagnosisOutput.innerHTML = ''; // Clear previous output
            getDiagnosisButton.disabled = true; // Disable button during processing

            try {
                let chatHistory = [];
                chatHistory.push({
                    role: "user",
                    parts: [{ text: `As a mobile car mechanic's AI assistant, analyze the following car issue description and suggest up to 3 potential services or common diagnoses. Be concise and professional. If the description is too vague, ask for more details. Issue: ${prompt}` }]
                });

                const payload = { contents: chatHistory };
                const apiKey = ""; // Canvas will provide this at runtime
                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

                const response = await fetch(apiUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });

                const result = await response.json();

                if (result.candidates && result.candidates.length > 0 &&
                    result.candidates[0].content && result.candidates[0].content.parts &&
                    result.candidates[0].content.parts.length > 0) {
                    const text = result.candidates[0].content.parts[0].text;
                    diagnosisOutput.innerHTML = `<p>${text.replace(/\n/g, '<br>')}</p>`; // Display response, handle newlines
                } else {
                    diagnosisOutput.innerHTML = '<p class="text-red-500">Could not get a diagnosis. Please try again or rephrase your issue.</p>';
                    console.error("Unexpected API response structure:", result);
                }
            } catch (error) {
                diagnosisOutput.innerHTML = '<p class="text-red-500">An error occurred while fetching diagnosis. Please try again later.</p>';
                console.error("Error calling Gemini API:", error);
            } finally {
                loadingSpinner.classList.add('hidden');
                getDiagnosisButton.disabled = false; // Re-enable button
            }
        });
    </script>
